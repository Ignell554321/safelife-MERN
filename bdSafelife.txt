CREATE TABLE Clinica(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  nombre VARCHAR(45),
  direccion VARCHAR(45),
  aforo INTEGER
);

CREATE TABLE Medico(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  nombre VARCHAR(45),
  apellido VARCHAR(45),
  dni CHAR(8),
  cmp VARCHAR(50),
  sede VARCHAR(150),
  imagen VARCHAR(200)

);

CREATE TABLE Especialidad(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  nombre VARCHAR(45)
);

CREATE TABLE Horario(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  rango VARCHAR(45),
  fecha VARCHAR(45)
);


CREATE TABLE Usuario(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  nombres VARCHAR(45),
  apellidos VARCHAR(45),
  dni VARCHAR(8),
  telefono VARCHAR(9),
  edad VARCHAR(3),
  email VARCHAR(200),
  password VARCHAR(150),
  direccion VARCHAR(250),
  distrito VARCHAR(100),
  provincia VARCHAR(100),
  departamento VARCHAR(100)
);

CREATE TABLE MedicoClinica (
  id     INTEGER AUTO_INCREMENT PRIMARY KEY, 
  id_clinica INTEGER, 
  id_medico INTEGER,
  FOREIGN KEY(id_clinica) REFERENCES Clinica(id),
  FOREIGN KEY(id_medico) REFERENCES Medico(id)
);

CREATE TABLE MedicoEspecialidad (
  id     INTEGER AUTO_INCREMENT PRIMARY KEY, 
  id_especialidad INTEGER, 
  id_medico INTEGER,
  FOREIGN KEY(id_especialidad) REFERENCES Especialidad(id),
  FOREIGN KEY(id_medico) REFERENCES Medico(id)
);

CREATE TABLE MedicoHorario(
  id     INTEGER AUTO_INCREMENT PRIMARY KEY, 
  id_medico INTEGER, 
  id_horario INTEGER,
  FOREIGN KEY(id_medico) REFERENCES Medico(id),
  FOREIGN KEY(id_horario) REFERENCES Horario(id)
);

CREATE TABLE Cita(
  id    INTEGER AUTO_INCREMENT PRIMARY KEY, 
  id_usuario INTEGER,
  rango VARCHAR(45),
  fecha VARCHAR(45),
  id_medico INTEGER,
  id_especialidad INTEGER,
  estado CHAR(1),
  monto DECIMAL(5,2),
  FOREIGN KEY(id_usuario) REFERENCES Usuario(id),
  FOREIGN KEY(id_medico) REFERENCES Medico(id),
  FOREIGN KEY(id_especialidad) REFERENCES Especialidad(id)
);

--QUERY
select * from Medico m inner join MedicoEspecialidad mes on (m.id=mes.id_medico)
inner join Especialidad e on (e.id=mes.id_especialidad) where e.id=1
